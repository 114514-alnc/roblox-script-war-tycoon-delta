-- [[ 全局变量控制 ]]
local LP = game.Players.LocalPlayer
local Camera = workspace.CurrentCamera
local Mouse = LP:GetMouse()
local RS = game:GetService("RunService")
local UIS = game:GetService("UserInputService")

_G.Aimbot = false
_G.AimRadius = 150
_G.Whitelist = {}
_G.NoRecoil = false
_G.ESP = false
_G.ESPColor = Color3.fromRGB(255, 0, 0)
_G.HitboxSize = 2
_G.ShowHitbox = false
_G.AutoFarm = false
_G.FarmFilter = "所有资源"
_G.AutoCapture = false
_G.ArmorTier = "轻型护甲"
_G.FlyEnabled = false
_G.FlySpeed = 70
_G.AutoRepair = false
_G.LowGraphic = false

-- 获取玩家列表
local function GetPlayerList()
    local p = {}
    for _, v in pairs(game.Players:GetPlayers()) do
        if v ~= LP then table.insert(p, v.Name) end
    end
    return p
end

-- 获取基地位置
local function GetBase()
    for _, t in pairs(workspace.Tycoon.Tycoons:GetChildren()) do
        if t:FindFirstChild("Owner") and t.Owner.Value == LP then return t end
    end
    return nil
end

-- [[ 核心逻辑循环 ]]

-- 1. 自瞄与视觉处理
RS.RenderStepped:Connect(function()
    if _G.Aimbot then
        local target = nil; local dist = _G.AimRadius
        for _, v in pairs(game.Players:GetPlayers()) do
            if v ~= LP and v.Team ~= LP.Team and v.Character and v.Character:FindFirstChild("Head") and not table.find(_G.Whitelist, v.Name) then
                local pos, onS = Camera:WorldToViewportPoint(v.Character.Head.Position)
                if onS then
                    local m = (Vector2.new(Mouse.X, Mouse.Y) - Vector2.new(pos.X, pos.Y)).Magnitude
                    if m < dist then target = v.Character.Head; dist = m end
                end
            end
        end
        if target then Camera.CFrame = CFrame.new(Camera.CFrame.Position, target.Position) end
    end
    
    for _, p in pairs(game.Players:GetPlayers()) do
        if p ~= LP and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local hrp = p.Character.HumanoidRootPart
            hrp.Size = Vector3.new(_G.HitboxSize, _G.HitboxSize, _G.HitboxSize)
            hrp.Transparency = _G.ShowHitbox and 0.7 or 1
            hrp.Color = _G.ESPColor
            if _G.ESP then
                local hl = p.Character:FindFirstChild("Highlight") or Instance.new("Highlight", p.Character)
                hl.FillColor = _G.ESPColor; hl.Enabled = true
            elseif p.Character:FindFirstChild("Highlight") then p.Character.Highlight.Enabled = false end
        end
    end
end)

-- 2. 枪械与飞行逻辑
task.spawn(function()
    local bv, bg
    while true do
        if _G.NoRecoil then
            for _, t in pairs(LP.Backpack:GetChildren()) do
                if t:IsA("Tool") and t:FindFirstChild("Settings") then
                    local s = require(t.Settings); s.Recoil = 0; s.Spread = 0
                end
            end
        end
        if _G.FlyEnabled then
            if not bv then
                bv = Instance.new("BodyVelocity", LP.Character.HumanoidRootPart)
                bv.MaxForce = Vector3.new(1e6, 1e6, 1e6)
                bg = Instance.new("BodyGyro", LP.Character.HumanoidRootPart)
                bg.MaxTorque = Vector3.new(1e6, 1e6, 1e6)
            end
            local dir = Vector3.new(0, 0.1, 0)
            if UIS:IsKeyDown(Enum.KeyCode.W) then dir = Camera.CFrame.LookVector end
            bv.Velocity = dir * _G.FlySpeed; bg.CFrame = Camera.CFrame
        elseif bv then
            bv:Destroy(); bg:Destroy(); bv = nil
        end
        task.wait()
    end
end)

-- 3. 自动采集与占点
task.spawn(function()
    while task.wait(3) do
        if _G.AutoFarm then
            for _, i in pairs(workspace:GetDescendants()) do
                if (_G.FarmFilter == "所有资源" and (i.Name == "OilDrum" or i.Name == "SupplyCrate")) then
                    LP.Character.HumanoidRootPart.CFrame = i.CFrame; task.wait(0.5)
                    firetouchinterest(LP.Character.HumanoidRootPart, i, 0); task.wait(0.5); break
                end
            end
        end
        if _G.AutoCapture then
            for _, cp in pairs(workspace:GetDescendants()) do
                if cp.Name == "CapturePoint" then LP.Character.HumanoidRootPart.CFrame = cp.CFrame; task.wait(8); break end
            end
        end
    end
end)
